CREATE DATABASE cloudapp;
USE cloudapp;
CREATE TABLE resource (
    res_id INT PRIMARY KEY,
    res_name VARCHAR(30),
    res_image VARCHAR(30)
);
CREATE TABLE resource_prop (
    res_prop_id INT PRIMARY KEY,
    res_id INT,
    prop_name VARCHAR(30),
    prop_input_type VARCHAR(30),
    is_mandatory VARCHAR(30),
    FOREIGN KEY (res_id) REFERENCES resource(res_id)
);

CREATE TABLE res_prop_list_value (
    id INT PRIMARY KEY,
    res_prop_id INT,
    prop_type VARCHAR(30),
    list_value VARCHAR(30),
    FOREIGN KEY (res_prop_id) REFERENCES resource_prop(res_prop_id)
);

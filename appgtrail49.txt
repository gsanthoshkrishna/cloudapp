function showCost(img) {
    const rect = img.getBoundingClientRect(); // Get the position of the image
    const resourceCostDisplay = document.getElementById('loadcontainer'); // Ensure you have an element with this ID

    resourceCostDisplay.style.top = `${rect.top - 30}px`;
    resourceCostDisplay.style.left = `${rect.left}px`;
    resourceCostDisplay.textContent = img.alt;
    resourceCostDisplay.style.display = 'block';

    // Send the image name to the server
    fetch('/get_resource_cost', {
        method: 'GET',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify({ resuid: img.alt }),
    })
    .then(response => response.json())
    .then(data => {
        console.log(data.message); // Log the server response
    })
    .catch(error => {
        console.error('Error:', error);
    });
}
